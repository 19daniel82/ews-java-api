43  /**
44   * Represents a response object created to remove a calendar item from a meeting
45   * cancellation.
46   */
47  @ServiceObjectDefinition(xmlElementName = XmlElementNames.RemoveItem, returnedByServer = false) public class RemoveFromCalendar extends
48                                                                                                                                  ServiceObject {
49  
50    /**
51     * The reference item.
52     */
53    private Item referenceItem;
54  
55    /**
56     * Initializes a new instance of the RemoveFromCalendar class.
57     *
58     * @param referenceItem The reference item
59     * @throws Exception the exception
60     */
61    public RemoveFromCalendar(Item referenceItem) throws Exception {
62      super(referenceItem.getService());
63  
64      referenceItem.throwIfThisIsNew();
65  
66      this.referenceItem = referenceItem;
67    }
68  
69    /**
70     * Internal method to return the schema associated with this type of object.
71     *
72     * @return The schema associated with this type of object.
73     */
74    @Override public ServiceObjectSchema getSchema() {
75      return ResponseObjectSchema.Instance;
76    }
77  
78    /**
79     * Gets the minimum required server version.
80     *
81     * @return Earliest Exchange version in which this service object type is
82     * supported.
83     */
84    @Override public ExchangeVersion getMinimumRequiredServerVersion() {
85      return ExchangeVersion.Exchange2007_SP1;
86    }
87  
88    /**
89     * Loads the specified set of property on the object.
90     *
91     * @param propertySet The property to load.
92     */
93    @Override
94    protected void internalLoad(PropertySet propertySet) {
95      throw new UnsupportedOperationException();
96    }
97  
98    /**
99     * Deletes the object.
100    *
101    * @param deleteMode              The deletion mode.
102    * @param sendCancellationsMode   Indicates whether meeting cancellation messages should be
103    *                                sent.
104    * @param affectedTaskOccurrences Indicate which occurrence of a recurring task should be
105    *                                deleted.
106    */
107   @Override
108   protected void internalDelete(DeleteMode deleteMode,
109       SendCancellationsMode sendCancellationsMode,
110       AffectedTaskOccurrence affectedTaskOccurrences) {
111     throw new UnsupportedOperationException();
112   }
113 
114   /**
115    * Create response object.
116    *
117    * @param parentFolderId     The parent folder id.
118    * @param messageDisposition The message disposition.
119    * @return A list of item that were created or modified as a results of
120    * this operation.
121    * @throws Exception the exception
122    */
123   public List<Item> internalCreate(FolderId parentFolderId, MessageDisposition messageDisposition) throws Exception {
124     ((ItemId) this.getPropertyBag().getObjectFromPropertyDefinition(
125         ResponseObjectSchema.ReferenceItemId))
126         .assign(this.referenceItem.getId());
127 
128     return this.getService().internalCreateResponseObject(this,
129         parentFolderId, messageDisposition);
130   }
