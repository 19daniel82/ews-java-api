122   /**
123    * Tries to read element from XML.
124    *
125    * @param reader accepts EwsServiceXmlReader
126    * @return True if element was read.
127    * @throws Exception throws Exception
128    */
129   @Override
130   public boolean tryReadElementFromXml(EwsServiceXmlReader reader)
131       throws Exception {
132     if (reader.getLocalName().equals(XmlElementNames.Bias)) {
133       this.delta = new TimeSpan((long)
134           reader.readElementValue(Integer.class) * 60 * 1000);
135       return true;
136     } else if (reader.getLocalName().equals(XmlElementNames.Time)) {
137       this.timeOfDay = TimeSpan.parse(reader.readElementValue());
138       return true;
139     } else if (reader.getLocalName().equals(XmlElementNames.DayOrder)) {
140       this.dayOrder = reader.readElementValue(Integer.class);
141       return true;
142     } else if (reader.getLocalName().equals(XmlElementNames.DayOfWeek)) {
143       this.dayOfTheWeek = reader.readElementValue(DayOfTheWeek.class);
144       return true;
145     } else if (reader.getLocalName().equals(XmlElementNames.Month)) {
146       this.month = reader.readElementValue(Integer.class);
147       return true;
148     } else if (reader.getLocalName().equals(XmlElementNames.Year)) {
149       this.year = reader.readElementValue(Integer.class);
150       return true;
151     } else {
152       return false;
153     }
154   }
155 
156   /**
157    * Writes the elements to XML.
158    *
159    * @param writer the writer
160    * @throws ServiceXmlSerializationException the service xml serialization exception
161    * @throws XMLStreamException the XML stream exception
162    */
163   @Override
164   public void writeElementsToXml(EwsServiceXmlWriter writer)
165       throws ServiceXmlSerializationException, XMLStreamException {
166     writer.writeElementValue(XmlNamespace.Types, XmlElementNames.Bias,
167         (int) this.delta.getMinutes());
168 
169     writer.writeElementValue(XmlNamespace.Types, XmlElementNames.Time,
170         EwsUtilities.timeSpanToXSTime(this.timeOfDay));
171 
172     writer.writeElementValue(XmlNamespace.Types, XmlElementNames.DayOrder,
173         this.dayOrder);
174 
175     writer.writeElementValue(XmlNamespace.Types, XmlElementNames.Month, this.month);
176 
177     // Only write DayOfWeek if this is a recurring time change
178     if (this.getYear() == 0) {
179       writer.writeElementValue(XmlNamespace.Types,
180           XmlElementNames.DayOfWeek, this.dayOfTheWeek);
181     }
182 
183     // Only emit year if it's non zero, otherwise AS returns
184     // "Request is invalid"
185     if (this.getYear() != 0) {
186       writer.writeElementValue(XmlNamespace.Types, XmlElementNames.Year,
187           this.getYear());
188     }
189   }

191   /**
192    * Gets if current time presents DST transition time
193    *
194    * @return month
195    */
196   protected boolean getHasTransitionTime() {
197     return this.month >= 1 && this.month <= 12;
198   }
