44    /**
45     * Convert Olson TimeZone to Microsoft TimeZone Generated using Unicode CLDR project Example:
46     * https://gist.github.com/scottmac/655675e9b4d4913c539c
47     *
48     * @param timeZone java timezone (Olson)
49     * @return a microsoft timezone identifier (ala Eastern Standard Time)
50     */
51    public static String getMicrosoftTimeZoneName(final TimeZone timeZone) {
52      if (timeZone == null) {
53        throw new IllegalArgumentException("Parameter \"timeZone\" must be defined");
54      }
55  
56      final String id = timeZone.getID();
57      return olsonTimeZoneToMs.get(id);
58    }
