24  package microsoft.exchange.webservices.data.notification;
25  
26  import microsoft.exchange.webservices.data.core.ExchangeService;
27  import microsoft.exchange.webservices.data.misc.AsyncCallback;
28  import microsoft.exchange.webservices.data.misc.IAsyncResult;
29  
30  /**
31   * Represents a pull subscription.
32   */
33  public final class PullSubscription extends SubscriptionBase {
34    /**
35     * The more events available.
36     */
37    private boolean moreEventsAvailable;
38  
39    /**
40     * Initializes a new instance.
41     *
42     * @param service the service
43     * @throws Exception the exception
44     */
45    public PullSubscription(ExchangeService service) throws Exception {
46      super(service);
47    }
48  
49    /**
50     * Obtains a collection of events that occurred on the subscribed folder
51     * since the point in time defined by the Watermark property. When GetEvents
52     * succeeds, Watermark is updated.
53     *
54     * @return Returns a collection of events that occurred since the last
55     * watermark
56     * @throws Exception the exception
57     */
58    public GetEventsResults getEvents() throws Exception {
59      GetEventsResults results = getService().getEvents(this.getId(),
60          this.getWaterMark());
61      this.setWaterMark(results.getNewWatermark());
62      this.moreEventsAvailable = results.isMoreEventsAvailable();
63      return results;
64    }
65  
66    /**
67     * Begins an asynchronous request to obtain a collection of events that occurred on the subscribed
68     * folder since the point in time defined by the Watermark property
69     *
70     * @param callback The AsyncCallback delegate
71     * @param state    An object that contains state information for this request
72     * @return An IAsyncResult that references the asynchronous request
73     * @throws Exception
74     */
75    public IAsyncResult beginGetEvents(AsyncCallback callback, Object state) throws Exception {
76      return this.getService().beginGetEvents(callback, state, this.getId(), this.getWaterMark());
77    }
78  
79    /**
80     * Ends an asynchronous request to obtain a collection of events that occurred on the subscribed
81     * folder since the point in time defined by the Watermark property. When EndGetEvents succeeds,
82     * Watermark is updated.
83     *
84     * @param asyncResult An IAsyncResult that references the asynchronous request.
85     * @return Returns a collection of events that occurred since the last watermark.
86     * @throws Exception
87     */
88    public GetEventsResults endGetEvents(IAsyncResult asyncResult) throws Exception {
89      GetEventsResults results = this.getService().endGetEvents(asyncResult);
90  
91      this.setWaterMark(results.getNewWatermark());
92      this.moreEventsAvailable = results.isMoreEventsAvailable();
93  
94      return results;
95    }
96  
97    /**
98     * Unsubscribes from the pull subscription.
99     *
100    * @throws Exception the exception
101    */
102   public void unsubscribe() throws Exception {
103     getService().unsubscribe(getId());
104   }
105 
106   /**
107    * Begins an asynchronous request to unsubscribe from the pull subscription.
108    *
109    * @param callback The AsyncCallback delegate.
110    * @param state    An object that contains state information for this request
111    * @return An IAsyncResult that references the asynchronous request
112    * @throws Exception
113    */
114   public IAsyncResult beginUnsubscribe(AsyncCallback callback, Object state) throws Exception {
115     return this.getService().beginUnsubscribe(callback, state, this.getId());
116   }
117 
118   /**
119    * Ends an asynchronous request to unsubscribe from the pull subscription.
120    *
121    * @param asyncResult An IAsyncResult that references the asynchronous request.
122    * @throws Exception
123    */
124   public void endUnsubscribe(IAsyncResult asyncResult) throws Exception {
125     this.getService().endUnsubscribe(asyncResult);
126   }
127 
128   /**
129    * Gets a value indicating whether more events are available on the server.
130    * MoreEventsAvailable is undefined (null) until GetEvents is called.
131    *
132    * @return true, if is more events available
133    */
134   public boolean isMoreEventsAvailable() {
135     return moreEventsAvailable;
136   }
137 }
