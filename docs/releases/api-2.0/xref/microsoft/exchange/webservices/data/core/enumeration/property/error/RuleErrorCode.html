24  package microsoft.exchange.webservices.data.core.enumeration.property.error;
25  
26  /**
27   * Defines the error codes identifying why a rule failed validation.
28   */
29  public enum RuleErrorCode {
30  
31    /**
32     * Active Directory operation failed.
33     */
34    ADOperationFailure,
35  
36    /**
37     * The e-mail account specified in the
38     * FromConnectedAccounts predicate was not found.
39     */
40    ConnectedAccountNotFound,
41  
42    /**
43     * The Rule object in a CreateInboxRuleOperation has an Id. The Ids of new
44     * rules are generated server side and
45     * should not be provided by the client.
46     */
47    CreateWithRuleId,
48  
49    /**
50     * The value is empty. An empty value is not allowed for the property.
51     */
52    EmptyValueFound,
53  
54    /**
55     * There already is a rule with the same priority.
56     */
57    DuplicatedPriority,
58  
59    /**
60     * There are multiple operations against the same rule.
61     * Only one operation per rule is allowed.
62     */
63    DuplicatedOperationOnTheSameRule,
64  
65    /**
66     * The folder does not exist in the user's mailbox.
67     */
68    FolderDoesNotExist,
69  
70    /**
71     * The e-mail address is invalid.
72     */
73    InvalidAddress,
74  
75    /**
76     * The date range is invalid.
77     */
78    InvalidDateRange,
79  
80    /**
81     * The folder Id is invalid.
82     */
83    InvalidFolderId,
84  
85    /**
86     * The size range is invalid.
87     */
88    InvalidSizeRange,
89  
90    /**
91     * The value is invalid.
92     */
93    InvalidValue,
94  
95    /**
96     * The message classification was not found.
97     */
98    MessageClassificationNotFound,
99  
100   /**
101    * No action was specified. At least one action must be specified.
102    */
103   MissingAction,
104 
105   /**
106    * The required parameter is missing.
107    */
108   MissingParameter,
109 
110   /**
111    * The range value is missing.
112    */
113   MissingRangeValue,
114 
115   /**
116    * The property cannot be modified.
117    */
118   NotSettable,
119 
120   /**
121    * The recipient does not exist.
122    */
123   RecipientDoesNotExist,
124 
125   /**
126    * The rule was not found.
127    */
128   RuleNotFound,
129 
130   /**
131    * The size is less than zero.
132    */
133   SizeLessThanZero,
134 
135   /**
136    * The string value is too big.
137    */
138   StringValueTooBig,
139 
140   /**
141    * The address is unsupported.
142    */
143   UnsupportedAddress,
144 
145   /**
146    * An unexpected error occured.
147    */
148   UnexpectedError,
149 
150   /**
151    * The rule is not supported.
152    */
153   UnsupportedRule
154 }
