020package microsoft.exchange.webservices.data.property.complex;
021
022import static org.mockito.Mockito.doReturn;
023
024import java.util.Date;
025
026import microsoft.exchange.webservices.data.property.complex.time.AbsoluteDateTransition;
027import microsoft.exchange.webservices.data.property.complex.time.TimeZoneDefinition;
028import microsoft.exchange.webservices.data.property.complex.time.TimeZoneTransition;
029
030import org.junit.Assert;
031import org.junit.Test;
032import org.junit.runner.RunWith;
033import org.junit.runners.JUnit4;
034import org.mockito.Mockito;
035
036@RunWith(JUnit4.class)
037public class TimeZoneTransitionCompareTest {
038
039  @Test
040  public void testAbsoluteDateTransitionsEqual() {
041    TimeZoneDefinition timeZoneDefinition = new TimeZoneDefinition();
042
043    Date date = new Date();
044
045    AbsoluteDateTransition first = Mockito.mock(AbsoluteDateTransition.class);
046    AbsoluteDateTransition second = Mockito.mock(AbsoluteDateTransition.class);
047
048    doReturn(date).when(first).getDateTime();
049    doReturn(date).when(second).getDateTime();
050
051    Assert.assertEquals(0, timeZoneDefinition.compare(first, second));
052  }
053
054  @Test
055  public void testAbsoluteDateTransitionsLess() {
056    TimeZoneDefinition timeZoneDefinition = new TimeZoneDefinition();
057
058    Date date1 = new Date();
059    Date date2 = new Date(date1.getTime() + 1);
060    
061    AbsoluteDateTransition first = Mockito.mock(AbsoluteDateTransition.class);
062    AbsoluteDateTransition second = Mockito.mock(AbsoluteDateTransition.class);
063
064    doReturn(date1).when(first).getDateTime();
065    doReturn(date2).when(second).getDateTime();
066
067    Assert.assertEquals(-1, timeZoneDefinition.compare(first, second));
068  }
069
070  @Test
071  public void testAbsoluteDateTransitionsGreater() {
072    TimeZoneDefinition timeZoneDefinition = new TimeZoneDefinition();
073
074    Date date1 = new Date();
075    Date date2 = new Date(date1.getTime() - 1);
076    
077    AbsoluteDateTransition first = Mockito.mock(AbsoluteDateTransition.class);
078    AbsoluteDateTransition second = Mockito.mock(AbsoluteDateTransition.class);
079
080    doReturn(date1).when(first).getDateTime();
081    doReturn(date2).when(second).getDateTime();
082
083    Assert.assertEquals(1, timeZoneDefinition.compare(first, second));
084  }
085
086  @Test
087  public void testAbsoluteDateTransitionAndTimeZoneTransition() {
088    TimeZoneDefinition timeZoneDefinition = new TimeZoneDefinition();
089
090    Date date1 = new Date();
091    
092    AbsoluteDateTransition first = Mockito.mock(AbsoluteDateTransition.class);
093    TimeZoneTransition second = Mockito.mock(TimeZoneTransition.class);
094
095    doReturn(date1).when(first).getDateTime();
096
097    Assert.assertEquals(1, timeZoneDefinition.compare(first, second));
098  }
099
100  @Test
101  public void testAbsoluteDateTransitionAndNull() {
102    TimeZoneDefinition timeZoneDefinition = new TimeZoneDefinition();
103   
104    Date date1 = new Date();   
105    AbsoluteDateTransition first = Mockito.mock(AbsoluteDateTransition.class);  
106    doReturn(date1).when(first).getDateTime();
107
108    Assert.assertEquals(1, timeZoneDefinition.compare(first, null));
109  }
110
111  @Test
112  public void testNullAndAbsoluteDateTransition() {
113    TimeZoneDefinition timeZoneDefinition = new TimeZoneDefinition();
114    
115    Date date1 = new Date();
116    AbsoluteDateTransition second = Mockito.mock(AbsoluteDateTransition.class);    
117    doReturn(date1).when(second).getDateTime();
118
119    Assert.assertEquals(-1, timeZoneDefinition.compare(null, second));
120  }
121
122  @Test
123  public void testCompareSameObject() {
124    TimeZoneDefinition timeZoneDefinition = new TimeZoneDefinition();
125
126    Date date1 = new Date();    
127    AbsoluteDateTransition first = Mockito.mock(AbsoluteDateTransition.class);
128    doReturn(date1).when(first).getDateTime();
129
130    Assert.assertEquals(0, timeZoneDefinition.compare(first, first));
131  }
