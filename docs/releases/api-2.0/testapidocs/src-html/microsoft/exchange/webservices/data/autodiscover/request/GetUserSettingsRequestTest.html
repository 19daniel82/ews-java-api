049/**
050 * Testclass for methods of GetUserSettingsRequest
051 */
052@RunWith(Parameterized.class)
053public class GetUserSettingsRequestTest extends BaseTest {
054
055  /**
056   * The ExchangeVersion which is under test
057   */
058  private final ExchangeVersion exchangeVersion;
059
060  /**
061   * The AutodiscoverService which is under test
062   */
063  private final AutodiscoverService autodiscoverService;
064
065  /**
066   * A mocked URI via HTTPS
067   */
068  private final URI uriMockHttps = URI.create("https://localhost");
069
070  /**
071   * A mocked URI via HTTP
072   */
073  private final URI uriMockHttp = URI.create("http://localhost");
074
075  /**
076   * Returns the Parameters which where handled to the constructor
077   *
078   * @return the available Services
079   * @throws ArgumentException
080   */
081  @Parameterized.Parameters
082  public static List<Object[]> getAutodiscoverServices() throws ArgumentException {
083    return new ArrayList<Object[]>() {
084
085      /**
086       * Constant serialized ID used for compatibility.
087       */
088      private static final long serialVersionUID = 1L;
089
090      {
091        for (ExchangeVersion exchangeVersion : ExchangeVersion.values()) {
092          add(new Object[] {exchangeVersion, new AutodiscoverService(exchangeVersion)});
093        }
094      }
095    };
096  }
097
098  /**
099   * Construct the Testobject with given Parameters
100   *
101   * @param exchangeVersion
102   * @param autodiscoverService
103   */
104  public GetUserSettingsRequestTest(final ExchangeVersion exchangeVersion,
105      final AutodiscoverService autodiscoverService) {
106    this.exchangeVersion = exchangeVersion;
107    this.autodiscoverService = autodiscoverService;
108  }
109
110  /**
111   * setup
112   */
113  @Before
114  public void setup() {
115    Assert.assertThat(this.exchangeVersion, IsNull.notNullValue());
116    Assert.assertThat(this.autodiscoverService, IsNull.notNullValue());
117    Assert.assertThat(uriMockHttp, IsNull.notNullValue());
118    Assert.assertThat(uriMockHttps, IsNull.notNullValue());
119  }
120
121  /**
122   * Nothing should be written to the OutputStream if expectPartnerToken is not set.
123   *
124   * @throws ServiceValidationException
125   * @throws XMLStreamException the XML stream exception
126   * @throws ServiceXmlSerializationException the service xml serialization exception
127   */
128  @Test
129  public void testWriteExtraCustomSoapHeadersToXmlWithoutPartnertoken()
130      throws ServiceValidationException, XMLStreamException, ServiceXmlSerializationException {
131    // HTTPS
132    GetUserSettingsRequest getUserSettingsRequest =
133        new GetUserSettingsRequest(autodiscoverService, uriMockHttps);
134
135    // Test without expected Partnertoken
136    ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
137    getUserSettingsRequest.writeExtraCustomSoapHeadersToXml(
138        new EwsServiceXmlWriter(exchangeServiceBaseMock, byteArrayOutputStream));
139
140    // nothing should be writyen to the outputstream
141    Assert.assertArrayEquals(byteArrayOutputStream.toByteArray(), new ByteArrayOutputStream().toByteArray());
142
143    // HTTP
144    getUserSettingsRequest = new GetUserSettingsRequest(autodiscoverService, uriMockHttp);
145
146    // Test without expected Partnertoken
147    byteArrayOutputStream = new ByteArrayOutputStream();
148    getUserSettingsRequest.writeExtraCustomSoapHeadersToXml(
149        new EwsServiceXmlWriter(exchangeServiceBaseMock, byteArrayOutputStream));
150
151    // nothing should be written to the outputstream
152    Assert.assertArrayEquals(byteArrayOutputStream.toByteArray(), new ByteArrayOutputStream().toByteArray());
153  }
154
155  /**
156   * Test if content is added correctly if expectPartnerToken is set.
157   *
158   * @throws ServiceValidationException
159   * @throws XMLStreamException the XML stream exception
160   * @throws ServiceXmlSerializationException the service xml serialization exception
161   */
162  @Test
163  public void testWriteExtraCustomSoapHeadersToXmlWithPartnertoken()
164      throws ServiceValidationException, XMLStreamException, ServiceXmlSerializationException {
165    GetUserSettingsRequest getUserSettingsRequest =
166        new GetUserSettingsRequest(autodiscoverService, uriMockHttps, Boolean.TRUE);
167
168    // Test without expected Partnertoken
169    ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
170    getUserSettingsRequest.writeExtraCustomSoapHeadersToXml(
171        new EwsServiceXmlWriter(exchangeServiceBaseMock, byteArrayOutputStream));
172
173    // data should be added the same way as mentioned
174    Assert.assertThat(byteArrayOutputStream.toByteArray(),
175        IsNot.not(new ByteArrayOutputStream().toByteArray()));
176
177    //TODO Test if the output is really correct
178  }
