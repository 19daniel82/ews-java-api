024package microsoft.exchange.webservices.data.misc;
025
026import microsoft.exchange.webservices.data.core.EwsUtilities;
027import org.apache.commons.codec.binary.Base64;
028import org.apache.commons.codec.binary.StringUtils;
029import org.junit.Assert;
030import org.junit.Test;
031import org.junit.runner.RunWith;
032import org.junit.runners.JUnit4;
033
034import java.util.Date;
035import java.util.UUID;
036
037@RunWith(JUnit4.class)
038public class IFunctionsTest {
039
040  @Test
041  public void testToString() {
042    final IFunctions.ToString f = IFunctions.ToString.INSTANCE;
043    Assert.assertEquals("null", f.func(null));
044    Assert.assertEquals("", f.func(""));
045    Assert.assertEquals("1", f.func(1));
046  }
047
048  @Test
049  public void testToBoolean() {
050    final IFunctions.ToBoolean f = IFunctions.ToBoolean.INSTANCE;
051    Assert.assertFalse(f.func(null));
052    Assert.assertFalse(f.func(""));
053    Assert.assertFalse(f.func("false"));
054    Assert.assertTrue(f.func("true"));
055  }
056
057  @Test
058  public void testStringToObject() {
059    final IFunctions.StringToObject f = IFunctions.StringToObject.INSTANCE;
060    Assert.assertNull(f.func(null));
061    Assert.assertEquals("", f.func(""));
062  }
063
064  @Test
065  public void testToUUID() {
066    final IFunctions.ToUUID f = IFunctions.ToUUID.INSTANCE;
067    try {
068      Assert.assertNull(f.func(null));
069    } catch (final Throwable ex) {
070      final UUID uuid = UUID.randomUUID();
071      Assert.assertEquals(uuid, f.func(uuid.toString()));
072    }
073  }
074
075  @Test
076  public void testBase64Decoder() {
077    final String value = "123";
078    final IFunctions.Base64Decoder f = IFunctions.Base64Decoder.INSTANCE;
079    Assert.assertArrayEquals(Base64.decodeBase64(value), (byte[]) f.func(value));
080  }
081
082  @Test
083  public void testBase64Encoder() {
084    final byte[] value = StringUtils.getBytesUtf8("123");
085    final IFunctions.Base64Encoder f = IFunctions.Base64Encoder.INSTANCE;
086    Assert.assertEquals(Base64.encodeBase64String(value), f.func(value));
087  }
088
089  @Test
090  public void testToLowerCase() {
091    final IFunctions.ToLowerCase f = IFunctions.ToLowerCase.INSTANCE;
092    Assert.assertNull(f.func(null));
093    Assert.assertEquals("", f.func(""));
094    Assert.assertEquals("abc", f.func("AbC"));
095  }
096
097  @Test
098  public void testDateTimeToXSDateTime() {
099    final IFunctions.DateTimeToXSDateTime f = IFunctions.DateTimeToXSDateTime.INSTANCE;
100    final Date value = new Date();
101    Assert.assertEquals(EwsUtilities.dateTimeToXSDateTime(value), f.func(value));
102  }
103
104}
