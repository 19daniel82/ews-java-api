020package microsoft.exchange.webservices.data.property.complex;
021
022import static org.junit.Assert.assertEquals;
023import static org.mockito.Mockito.doReturn;
024import microsoft.exchange.webservices.data.core.EwsServiceXmlReader;
025import microsoft.exchange.webservices.data.core.XmlElementNames;
026import microsoft.exchange.webservices.data.property.complex.recurrence.pattern.Recurrence.MonthlyPattern;
027import microsoft.exchange.webservices.data.property.complex.recurrence.pattern.Recurrence.YearlyPattern;
028
029import org.junit.Test;
030import org.mockito.Mockito;
031
032public class RecurrenceReaderTest {
033
034  @Test
035  public void testMonthlyPattern() throws Exception {
036
037    EwsServiceXmlReader reader = Mockito.mock(EwsServiceXmlReader.class);
038    doReturn(XmlElementNames.DayOfMonth).when(reader).getLocalName();
039    doReturn(1).when(reader).readElementValue(Integer.class);
040
041    MonthlyPattern monthly = new MonthlyPattern();
042    monthly.tryReadElementFromXml(reader);
043
044    assertEquals(1, monthly.getDayOfMonth());
045  }
046
047  @Test
048  public void testYearlyPattern() throws Exception {
049
050    EwsServiceXmlReader reader = Mockito.mock(EwsServiceXmlReader.class);
051    doReturn(XmlElementNames.DayOfMonth).when(reader).getLocalName();
052    doReturn(1).when(reader).readElementValue(Integer.class);
053
054    YearlyPattern yearly = new YearlyPattern();
055    yearly.tryReadElementFromXml(reader);
056
057    assertEquals(1, yearly.getDayOfMonth());
058  }
059
060}
