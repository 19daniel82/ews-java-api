024package microsoft.exchange.webservices.data.core;
025
026import microsoft.exchange.webservices.data.core.service.ServiceObject;
027import microsoft.exchange.webservices.data.core.service.item.Item;
028import microsoft.exchange.webservices.data.core.enumeration.misc.ExchangeVersion;
029import microsoft.exchange.webservices.data.core.exception.misc.ArgumentException;
030import microsoft.exchange.webservices.data.core.exception.service.local.ServiceObjectPropertyException;
031import microsoft.exchange.webservices.data.misc.OutParam;
032import microsoft.exchange.webservices.data.property.definition.IntPropertyDefinition;
033import microsoft.exchange.webservices.data.property.definition.RecurrencePropertyDefinition;
034import org.junit.Test;
035import org.junit.runner.RunWith;
036import org.junit.runners.JUnit4;
037
038@RunWith(JUnit4.class)
039public class PropertyBagTest {
040
041  /**
042   * Calling tryGetPropertyType with invalid data.
043   * Expecting exception
044   * 
045   * @throws Exception
046   */
047  @Test(expected=ArgumentException.class)
048  public void tryGetPropertyType() throws Exception{
049    PropertyBag pb = createPropertyBag();
050    pb.tryGetPropertyType(String.class, new RecurrencePropertyDefinition("test", "none", null, ExchangeVersion.Exchange2010_SP2), new OutParam<String>());
051  }
052
053  @Test(expected = ServiceObjectPropertyException.class)
054  public void testGetObjectFromPropertyDefinition() throws Exception {
055    PropertyBag pb = createPropertyBag();
056    pb.getObjectFromPropertyDefinition(new IntPropertyDefinition("", "none", ExchangeVersion.Exchange2007_SP1));
057  }
058
059
060  private PropertyBag createPropertyBag() throws Exception {
061    ExchangeService es = new ExchangeService();
062    ServiceObject owner = new Item(es);
063    return new PropertyBag(owner);
064  }
065
066}
