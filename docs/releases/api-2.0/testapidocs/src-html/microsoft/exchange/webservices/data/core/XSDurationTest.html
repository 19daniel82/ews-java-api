020package microsoft.exchange.webservices.data.core;
021
022import microsoft.exchange.webservices.data.core.EwsUtilities;
023import microsoft.exchange.webservices.data.misc.TimeSpan;
024
025import org.junit.Assert;
026import org.junit.Test;
027import org.junit.runner.RunWith;
028import org.junit.runners.JUnit4;
029
030@RunWith(JUnit4.class)
031public class XSDurationTest {
032
033  // Tests for EwsUtilities.getXSDurationToTimeSpan()
034
035  @Test
036  public void testPeriodHours() {
037    TimeSpan timeSpan = EwsUtilities.getXSDurationToTimeSpan("-PT13H");
038    Assert.assertEquals("-P0DT13H0M0.0S", EwsUtilities.getTimeSpanToXSDuration(timeSpan));
039  }
040
041  @Test
042  public void testPeriodHoursMinutes() {
043    TimeSpan timeSpan = EwsUtilities.getXSDurationToTimeSpan("-PT5H30M");
044    Assert.assertEquals("-P0DT5H30M0.0S", EwsUtilities.getTimeSpanToXSDuration(timeSpan));
045  }
046
047  @Test
048  public void testPeriodFull() {
049    TimeSpan timeSpan = EwsUtilities.getXSDurationToTimeSpan("PT2H30M59.0S");
050    Assert.assertEquals("P0DT2H30M59.0S", EwsUtilities.getTimeSpanToXSDuration(timeSpan));
051  }
052
053  @Test
054  public void testPeriodFullNegative() {
055    TimeSpan timeSpan = EwsUtilities.getXSDurationToTimeSpan("-PT2H30M59.0S");
056    Assert.assertEquals("-P0DT2H30M59.0S", EwsUtilities.getTimeSpanToXSDuration(timeSpan));
057  }
058
059  @Test
060  public void testPeriodFail2() {
061    TimeSpan timeSpan = EwsUtilities.getXSDurationToTimeSpan("PT2H100M59.0S");
062    Assert.assertEquals("P0DT3H40M59.0S", EwsUtilities.getTimeSpanToXSDuration(timeSpan));
063  }
064
065  @Test(expected = IllegalArgumentException.class)
066  public void testPeriodFail() {
067    TimeSpan timeSpan = EwsUtilities.getXSDurationToTimeSpan("P2H30M59.0S");
068    Assert.assertEquals("-P0DT2H30M59.0S", EwsUtilities.getTimeSpanToXSDuration(timeSpan));
069  }
070
071}
