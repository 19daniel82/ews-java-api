024package microsoft.exchange.webservices.data.core.response;
025
026import microsoft.exchange.webservices.data.core.EwsServiceXmlReader;
027import microsoft.exchange.webservices.data.core.EwsUtilities;
028import microsoft.exchange.webservices.data.core.XmlAttributeNames;
029import microsoft.exchange.webservices.data.core.XmlElementNames;
030import microsoft.exchange.webservices.data.core.enumeration.misc.XmlNamespace;
031import microsoft.exchange.webservices.data.core.exception.service.local.ServiceLocalException;
032import microsoft.exchange.webservices.data.misc.id.AlternateId;
033import microsoft.exchange.webservices.data.misc.id.AlternateIdBase;
034import microsoft.exchange.webservices.data.misc.id.AlternatePublicFolderId;
035import microsoft.exchange.webservices.data.misc.id.AlternatePublicFolderItemId;
036
037/**
038 * Represents the response to an individual Id conversion operation.
039 */
040public final class ConvertIdResponse extends ServiceResponse {
041
042  /**
043   * The converted id.
044   */
045  private AlternateIdBase convertedId;
046
047  /**
048   * Initializes a new instance of the class.
049   */
050  public ConvertIdResponse() {
051    super();
052  }
053
054  /**
055   * Reads response elements from XML.
056   *
057   * @param reader the reader
058   * @throws InstantiationException the instantiation exception
059   * @throws IllegalAccessException the illegal access exception
060   * @throws ServiceLocalException  the service local exception
061   * @throws Exception              the exception
062   */
063  @Override
064  protected void readElementsFromXml(EwsServiceXmlReader reader)
065      throws InstantiationException, IllegalAccessException, ServiceLocalException, Exception {
066    super.readElementsFromXml(reader);
067    reader.readStartElement(XmlNamespace.Messages,
068        XmlElementNames.AlternateId);
069    String alternateIdClass = reader.readAttributeValue(
070        XmlNamespace.XmlSchemaInstance, XmlAttributeNames.Type);
071
072    int aliasSeparatorIndex = alternateIdClass.indexOf(':');
073
074    if (aliasSeparatorIndex > -1) {
075      alternateIdClass = alternateIdClass
076          .substring(aliasSeparatorIndex + 1);
077    }
078
079    // Alternate Id classes are responsible fro reading the AlternateId end
080    // element when necessary
081    if (alternateIdClass.equals(AlternateId.SchemaTypeName)) {
082      this.convertedId = new AlternateId();
083    } else if (alternateIdClass
084        .equals(AlternatePublicFolderId.SchemaTypeName)) {
085      this.convertedId = new AlternatePublicFolderId();
086    } else if (alternateIdClass
087        .equals(AlternatePublicFolderItemId.SchemaTypeName)) {
088      this.convertedId = new AlternatePublicFolderItemId();
089    } else {
090      EwsUtilities
091          .ewsAssert(false, "ConvertIdResponse.ReadElementsFromXml",
092                     String.format("Unknown alternate Id class: %s", alternateIdClass));
093    }
094
095    this.convertedId.loadAttributesFromXml(reader);
096    reader.readEndElement(XmlNamespace.Messages,
097        XmlElementNames.AlternateId);
098  }
099
100  /**
101   * Reads response elements from XML.
102   *
103   * @return the converted id
104   */
105  public AlternateIdBase getConvertedId() {
106    return this.convertedId;
107  }
108
109}
